nginx config setup :
sudo apt purge nginx
sudo apt purge nginx-common
sudo apt purge nginx-full

sudo apt autoremove

// specifically for those who want to reset all the main nginx configurations
sudo rm -rf /etc/nginx
sudo rm -rf /var/log/nginx

// specifically for those who want to reset all data or webserver structure
sudo find /var/www -mindepth 1 ! -name 'html' -exec rm -rf {} +

// specifically for those who want to reset all webserver configurations
sudo find /etc/nginx/sites-available -type f ! -name 'default' -delete // delete all webserver config except config "default"
sudo find /etc/nginx/sites-enabled -type f ! -name 'default' -delete // delete all webserver config except config "default"

// specifically if you want to reset the individual webserver configuration
sudo rm /etc/nginx/sites-enabled/[site]
sudo rm /etc/nginx/sites-available/[site]

// check the remaining nginx dependencies again
systemctl list-unit-files | grep nginx

find / -name "*nginx*" 2>/dev/null

dpkg -l | grep nginx
