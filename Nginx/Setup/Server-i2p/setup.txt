nginx config setup :
cd /etc/nginx/

sudo nano nginx.conf
^ config :
http {
        ##
        # I2P settings
        ##

        limit_req_zone $binary_remote_addr zone=i2p:10m rate=5r/s;
}

cat nginx.conf

cd sites-available

sudo mkdir -p i2p-site

cd i2p-site

sudo nano i2p-proxy-[id]
^ config :
server {
    listen 127.0.0.1:[port]; // adjust to the tunnel port on the i2p webserver
    server_name localhost;

    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";

    server_tokens off;

    location / {
        proxy_pass http://127.0.0.1:[port]; // Set the port of the webserver you created to be directed to the i2p webserver tunnel

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP 127.0.0.1;
        proxy_set_header X-Forwarded-For 127.0.0.1;

        proxy_http_version 1.1;

        limit_req zone=i2p burst=10;
    }

    if ($request_method !~ ^(GET|HEAD|POST)$ ) {
        return 444;
    }

    location ~ /\.(git|env|htaccess) {
        deny all;
    }
}

cat i2p-proxy-[id]

sudo nginx -t

sudo ln -s /etc/nginx/sites-available/i2p-site/i2p-proxy-[id] /etc/nginx/sites-enabled/

ls -l /etc/nginx/sites-available/i2p-site

ls /etc/nginx/sites-available/
ls /etc/nginx/sites-enabled/

sudo systemctl reload nginx
sudo systemctl restart nginx
sudo systemctl status nginx

sudo curl http://127.0.0.1:[port]

sudo curl http://localhost:[port]
