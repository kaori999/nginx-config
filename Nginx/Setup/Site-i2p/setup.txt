nginx config setup :
cd /var/www

sudo mkdir -p i2p-page

cd i2p-page

sudo mkdir -p site-i2p-[id]

sudo touch site-i2p-[id]/index.html

ls -l /var/www/
ls -l /var/www/i2p-page
ls -l /var/www/i2p-page/site-i2p-[id]

cd /etc/nginx/sites-available

cd i2p-site

sudo nano i2p-site-[id]
^ config :
server {
    listen 127.0.0.1:[port];
    server_name _;

    root /var/www/i2p-page/site-i2p-[id];
    index index.html;

    access_log /var/log/nginx/site-i2p-[id]_access.log;
    error_log  /var/log/nginx/site-i2p-[id]_error.log;

    add_header Referrer-Policy no-referrer;
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";

    location / {
        try_files $uri $uri/ =404;
    }

    location ~ /\. {
        deny all;
    }
}

cat i2p-site-[id]

sudo nginx -t

sudo ln -s /etc/nginx/sites-available/i2p-site/i2p-site-[id] /etc/nginx/sites-enabled/

ls -l /etc/nginx/sites-available/i2p-site

ls /etc/nginx/sites-available/
ls /etc/nginx/sites-enabled/

sudo systemctl reload nginx
sudo systemctl restart nginx
sudo systemctl status nginx

sudo curl http://127.0.0.1:[port]

sudo curl http://localhost:[port]

ls -l /var/log/nginx
ls -ld /var/log/nginx

sudo cat /var/log/nginx/site-i2p-[id]_error.log
sudo cat /var/log/nginx/site-i2p-[id]_access.log
